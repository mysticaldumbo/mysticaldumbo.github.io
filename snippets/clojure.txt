(ns my-namespace
  (:gen-class))

(defn greet [name]
  "This function greets the user with the given name."
  (println (str "Hello, " name "!")))

(defn -main [& args]
  "This is the main function."
  (if (empty? args)
    ;; If no arguments are provided, print a usage message
    (println "Usage: Please provide a name.")
    ;; Otherwise, call the greet function with the first argument
    (greet (first args))))
