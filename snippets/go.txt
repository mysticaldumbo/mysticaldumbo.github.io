package main

import "fmt"

func isPrime(num int) bool {
    if num <= 1 {
        return false
    }
    for i := 2; i*i <= num; i++ {
        if num%i == 0 {
            return false
        }
    }
    return true
}

func generatePrimes(n int) []int {
    primes := []int{}
    num := 2
    for len(primes) < n {
        if isPrime(num) {
            primes = append(primes, num)
        }
        num++
    }
    return primes
}

func main() {
    n := 10
    // Print first n prime numbers
    fmt.Printf("First %d prime numbers: %v\n", n, generatePrimes(n))
}
