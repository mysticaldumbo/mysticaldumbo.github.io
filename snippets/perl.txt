#!/usr/bin/perl

use strict;
use warnings;

# Define a function to calculate the factorial of a number
sub factorial {
    my $n = shift; # Get the number argument
    return 1 if $n == 0 or $n == 1; # Base case: factorial of 0 or 1 is 1
    return $n * factorial($n - 1); # Recursive case: multiply n by factorial of (n - 1)
}

# Entry point of the program
sub main {
    # Calculate and print the factorial of 5
    my $result = factorial(5);
    print "Factorial of 5 is: $result\n";
}

# Call the main function to start the program
main();
