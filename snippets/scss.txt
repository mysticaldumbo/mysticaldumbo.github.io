// Define variables
$primary-color: #007bff; // Primary color
$secondary-color: #6c757d; // Secondary color

// Define mixins
@mixin border-radius($radius: 5px) { // Mixin for border radius with a default value of 5px
  border-radius: $radius;
}

@mixin box-shadow($x: 0, $y: 0, $blur: 5px, $color: rgba(0, 0, 0, 0.3)) { // Mixin for box shadow with default values
  box-shadow: $x $y $blur $color;
}

// Define styles
.button {
  background-color: $primary-color; // Use primary color as background
  color: white; // Text color
  padding: 10px 20px; // Padding
  @include border-radius; // Apply border radius mixin
  @include box-shadow(2px, 2px); // Apply box shadow mixin with custom values
}

// Usage
.button:hover {
  background-color: darken($primary-color, 10%); // Darken primary color on hover
}
