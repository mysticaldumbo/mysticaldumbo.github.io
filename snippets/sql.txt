-- Create a table to store information about employees
CREATE TABLE employees (
    employee_id INT PRIMARY KEY,
    first_name VARCHAR(50),
    last_name VARCHAR(50),
    department VARCHAR(50),
    salary DECIMAL(10, 2)
);

-- Insert sample data into the employees table
INSERT INTO employees (employee_id, first_name, last_name, department, salary)
VALUES
    (1, 'John', 'Doe', 'IT', 50000.00),
    (2, 'Jane', 'Smith', 'HR', 60000.00),
    (3, 'Alice', 'Johnson', 'Marketing', 55000.00);

-- Retrieve all employees from the IT department
SELECT * FROM employees WHERE department = 'IT';

-- Update Jane Smith's salary
UPDATE employees SET salary = 65000.00 WHERE first_name = 'Jane' AND last_name = 'Smith';

-- Delete employee with ID 3
DELETE FROM employees WHERE employee_id = 3;
