interface User {
    name: string;
    age: number;
}

// Define a functional component named Greeting
const Greeting: React.FC<User> = ({ name, age }) => {
    return (
        <div>
            <h1>Hello, {name}!</h1>
            <p>You are {age} years old.</p>
        </div>
    );
};

// Define an example user
const user: User = {
    name: "John",
    age: 30
};

// Render the Greeting component with the example user
ReactDOM.render(<Greeting {...user} />, document.getElementById("root"));
